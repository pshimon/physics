\section{Introduction}

Nelson stochastic mechanics reproduces the Schr\"odinger equation
from diffusion processes. However, the original formulation
describes only scalar quantum mechanics.
The purpose of this paper is to construct spin geometrically
within the stochastic framework.

We show that spin emerges from:

\begin{enumerate}
\item rotational corrections to drift,
\item internal Brownian motion on compact Lie groups,
\item principal bundle holonomy,
\item relativistic spin-bundle lifting,
\item non-Abelian compact group diffusion.
\end{enumerate}

\section{Scalar Nelson Theory Revisited}

Let $X_t \in \mathbb{R}^3$ satisfy

\[
dX_t = b(X_t,t) dt + \sqrt{\frac{\hbar}{m}}\, dW_t.
\]

Forward/backward derivatives:

\[
D_\pm f = \partial_t f + b_\pm\cdot\nabla f
\pm \frac{\hbar}{2m}\Delta f.
\]

Define

\[
v = \frac{b_+ + b_-}{2}, \qquad
u = \frac{b_+ - b_-}{2}.
\]

\begin{lemma}
Consistency implies
\[
u = \frac{\hbar}{2m}\nabla\ln\rho.
\]
\end{lemma}

Mean acceleration:

\[
a = \frac12(D_+D_- + D_-D_+)X.
\]

Newton law:

\[
ma = -\nabla V.
\]

After algebra:

\[
\partial_t S + \frac{(\nabla S)^2}{2m}
+ V
- \frac{\hbar^2}{2m}
\frac{\Delta\sqrt{\rho}}{\sqrt{\rho}}
=0.
\]

\begin{theorem}
Scalar Nelson diffusion yields the Schr\"odinger equation.
\end{theorem}

\section{Pauli Equation: Full Stochastic Derivation}

Let $\Psi\in\mathbb{C}^2$.

Define:

\[
\rho=\Psi^\dagger\Psi,
\quad
s^i=\Psi^\dagger\sigma^i\Psi.
\]

Pauli current:

\[
j=
\frac{\hbar}{m}\Im(\Psi^\dagger\nabla\Psi)
-\frac{e}{m}A\rho
+\frac{\hbar}{2m}\nabla\times s.
\]

Drift:

\[
v=\frac{j}{\rho}.
\]

Stochastic Lagrangian:

\[
L=
\frac{m}{2}(v^2+u^2)
- e\phi
+ \frac{e}{2m}s\cdot B.
\]

Variation with respect to $\Psi^\dagger$:

\[
\delta \mathcal{A}=0.
\]

After careful integration by parts and using identity

\[
(\sigma\cdot\Pi)^2
=
\Pi^2 - e\hbar\sigma\cdot B,
\]

we obtain:

\[
i\hbar\partial_t\Psi
=
\left[
\frac{1}{2m}(\sigma\cdot(-i\hbar\nabla-eA))^2
+ e\phi
\right]\Psi.
\]

\begin{theorem}
The stochastic variational principle reproduces the Pauli equation.
\end{theorem}

\section{$SU(2)$ Brownian Motion: Explicit Coordinates}

Parameterize $SU(2)$ by Euler angles $(\alpha,\beta,\gamma)$.

Left-invariant vector fields:

\[
J_1 = \sin\gamma \partial_\beta
+ \cot\beta\cos\gamma\partial_\gamma
- \frac{\cos\gamma}{\sin\beta}\partial_\alpha,
\]

(similar expressions for $J_2,J_3$).

Laplace–Beltrami operator:

\[
\Delta_{SU(2)} = J_1^2+J_2^2+J_3^2.
\]

Internal SDE:

\[
dg_t = g_t \circ d\Xi_t,
\quad
d\Xi_t = \sum_a T_a \circ dW_t^a.
\]

Generator:

\[
\mathcal{L} =
\frac{\hbar}{2m}\Delta_x
+
\kappa\Delta_{SU(2)}.
\]

\begin{theorem}
Eigenfunctions of $\Delta_{SU(2)}$ correspond to
irreducible spin-$j$ representations.
\end{theorem}

Restrict to $j=1/2$ sector → Pauli spinors.

\section{Relativistic Proper-Time Diffusion}

Let $X^\mu(\tau)$ satisfy

\[
dX^\mu = b^\mu d\tau
+ \sqrt{\frac{\hbar}{m}} dW^\mu,
\]

with

\[
b^\mu b_\mu = 1.
\]

Introduce spin connection $\omega_\mu$.

Covariant derivative:

\[
D_\mu=\partial_\mu+\omega_\mu.
\]

Define stochastic relativistic action:

\[
\mathcal{A}
=
E\int d\tau
\left[
\frac{m}{2}U^\mu U_\mu
- m
\right].
\]

After enforcing covariance and representation of Clifford algebra:

\[
\{\gamma^\mu,\gamma^\nu\}=2\eta^{\mu\nu}.
\]

\begin{theorem}
Lorentz-covariant stochastic diffusion in spin bundle
implies Dirac equation:
\[
(i\gamma^\mu D_\mu - m)\Psi=0.
\]
\end{theorem}
