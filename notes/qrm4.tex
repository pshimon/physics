\section{Semiclassical and Microlocal Analysis of Magnetic Systems}

\subsection{Magnetic Classical Flow}

Let the Hamiltonian on $(T^*M,\omega_B)$ be:
\[
H(x,p)=\frac{1}{2m} g^{ij}(x)p_i p_j.
\]

Hamiltonian vector field $X_H$ is defined by:
\[
\iota_{X_H} \omega_B = dH.
\]

In local coordinates one finds:
\begin{align}
\dot{x}^i &= \frac{\partial H}{\partial p_i} = \frac{1}{m} g^{ij} p_j, \\
\dot{p}_i &= -\frac{\partial H}{\partial x^i} + B_{ij} \dot{x}^j.
\end{align}

Thus trajectories are \emph{magnetic geodesics}.

\subsection{Semiclassical Quantization}

Define semiclassical parameter $\hbar \to 0$ and consider operator:
\[
\hat{H}_\hbar = \frac{1}{2m}(-i\hbar\nabla - A)^2.
\]

\begin{definition}
The semiclassical principal symbol is:
\[
\sigma(\hat{H}_\hbar)(x,p) = \frac{1}{2m} g^{ij}(x) (p_i - A_i)(p_j - A_j).
\]
\end{definition}

\subsection{Magnetic Egorov Theorem}

\begin{theorem}[Magnetic Egorov]
Let $U_t = e^{-it\hat{H}_\hbar/\hbar}$. Then for any pseudodifferential operator $Op_\hbar(a)$:
\[
U_t^* Op_\hbar(a) U_t
=
Op_\hbar(a \circ \Phi_t) + O(\hbar),
\]
where $\Phi_t$ is the magnetic Hamiltonian flow generated by $H$ with symplectic form $\omega_B$.
\end{theorem}

\begin{proof}
The proof follows standard semiclassical calculus. The key modification is that the Poisson bracket is computed with respect to $\omega_B$, leading to:
\[
\{a,H\}_B = \partial_{x^i} a \, \partial_{p_i} H - \partial_{p_i} a \, \partial_{x^i} H + B_{ij} \partial_{p_i} a \, \partial_{p_j} H.
\]
The Heisenberg evolution satisfies:
\[
\frac{d}{dt} Op_\hbar(a_t) = \frac{i}{\hbar}[\hat{H}_\hbar, Op_\hbar(a_t)],
\]
which matches transport along $\Phi_t$ modulo $O(\hbar)$.
\end{proof}

\subsection{Magnetic Gutzwiller Trace Formula}

Consider the trace:
\[
\mathrm{Tr}\, e^{-it\hat{H}/\hbar}.
\]

\begin{theorem}[Magnetic Trace Formula]
In the semiclassical limit:
\[
\mathrm{Tr}\, e^{-it\hat{H}/\hbar}
\sim
\sum_{\gamma}
A_\gamma
\exp\left(
\frac{i}{\hbar} S_\gamma + i\mu_\gamma \frac{\pi}{2}
\right),
\]
where:
\begin{itemize}
\item $\gamma$ are periodic magnetic geodesics,
\item $S_\gamma = \oint_\gamma (p\,dx + A)$,
\item $\mu_\gamma$ is Maslov index,
\item $A_\gamma$ depends on stability.
\end{itemize}
\end{theorem}

\begin{proof}[Sketch]
Use stationary phase approximation of the path integral. The action functional is:
\[
S[x] = \int \left( \frac{m}{2} g_{ij}\dot{x}^i \dot{x}^j + A_i \dot{x}^i \right) dt.
\]
Critical points are magnetic geodesics. The quadratic fluctuation yields determinant factor giving $A_\gamma$.
\end{proof}

\section{Full Stochastic Derivation (Nelson Mechanics with Gauge Field)}

\subsection{Diffusion on Riemannian Manifold}

Let $(M,g)$ be complete. Consider diffusion process $X_t$ satisfying:
\[
dX_t^i = b^i(X_t,t)\,dt + \sqrt{\frac{\hbar}{m}}\, dW_t^i,
\]
where $W_t$ is Brownian motion with generator:
\[
\frac{1}{2}\Delta_g.
\]

Thus generator of $X_t$ is:
\[
\mathcal{L} = b^i \nabla_i + \frac{\hbar}{2m}\Delta_g.
\]

---

\subsection{Forward and Backward Derivatives}

Define:
\begin{align}
D f &= \lim_{dt\to0^+} \mathbb{E}\left[\frac{f(X_{t+dt}) - f(X_t)}{dt}\right],\\
D_* f &= \lim_{dt\to0^+} \mathbb{E}\left[\frac{f(X_t) - f(X_{t-dt})}{dt}\right].
\end{align}

One finds:
\begin{align}
D f &= \partial_t f + b^i \nabla_i f + \frac{\hbar}{2m}\Delta f,\\
D_* f &= \partial_t f + b_*^i \nabla_i f - \frac{\hbar}{2m}\Delta f.
\end{align}

---

\subsection{Osmotic and Current Velocities}

Define:
\[
v = \frac{b + b_*}{2}, \quad u = \frac{b - b_*}{2}.
\]

\begin{proposition}
\[
u = \frac{\hbar}{2m} \nabla \ln \rho.
\]
\end{proposition}

\begin{proof}
From forward/backward Fokker--Planck equations:
\[
\partial_t \rho + \nabla_i(\rho b^i) = \frac{\hbar}{2m}\Delta \rho,
\]
\[
\partial_t \rho + \nabla_i(\rho b_*^i) = -\frac{\hbar}{2m}\Delta \rho.
\]
Subtracting gives:
\[
\nabla_i(\rho u^i) = \frac{\hbar}{2m}\Delta \rho,
\]
which implies the stated result.
\end{proof}

\subsection{Continuity Equation}

Adding the two equations yields:
\[
\partial_t \rho + \nabla_i(\rho v^i) = 0.
\]

a^i = \partial_t v^i + (v\cdot \nabla)v^i - (u\cdot \nabla)u^i - \frac{\hbar}{2m}\Delta u^i.

\subsection{Derivation of Quantum Hamilton--Jacobi Equation}

Substitute:
\[
v = \frac{1}{m}(\nabla S - A), \quad
u = \frac{\hbar}{2m}\nabla \ln \rho.

After algebra (expanding all terms), one obtains:
\[
\partial_t S
+
\frac{1}{2m}|\nabla S - A|^2
+ V
- \frac{\hbar^2}{2m}\frac{\Delta \sqrt{\rho}}{\sqrt{\rho}}

\subsection{Reconstruction of Schr\"odinger Equation}

Define:
\[
\psi = \sqrt{\rho} e^{iS/\hbar}.
\]

\begin{theorem}
$\psi$ satisfies:
\[
i\hbar \partial_t \psi
=
\frac{1}{2m}(-i\hbar\nabla - A)^2 \psi + V\psi.
\]
\end{theorem}

\begin{proof}
Compute derivatives explicitly:
\[
\nabla \psi = \left( \frac{\nabla \rho}{2\rho} + \frac{i}{\hbar}\nabla S \right)\psi.
\]

Insert into covariant Laplacian and combine with continuity equation and Hamilton--Jacobi equation. All nonlinear terms cancel.
\end{proof}
