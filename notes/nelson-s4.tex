\section{Mathematical Preliminaries}

Let $(\Omega,\mathcal{F},\mathbb{P})$ be a filtered probability space
supporting independent Wiener processes $W^i_t$.

We work with diffusion processes solving SDEs of It\^o form
\[
dX_t = b(X_t,t)dt + \sigma dW_t.
\]

Generator:
\[
\mathcal{L}f = b\cdot\nabla f + \frac12\sigma^2 \Delta f.
\]

Forward/backward mean derivatives:
\[
D_\pm f = \partial_t f + b_\pm\cdot\nabla f
\pm \frac{\sigma^2}{2}\Delta f.
\]

All generators are defined on $C_c^\infty$ and closed in $L^2$.

\section{Scalar Nelson Theory (Rigorous Form)}

Set $\sigma^2 = \hbar/m$.

Define current and osmotic velocities:

\[
v = \frac{b_+ + b_-}{2},
\quad
u = \frac{b_+ - b_-}{2}.
\]

\begin{lemma}
Consistency of forward/backward Fokker–Planck equations implies
\[
u = \frac{\hbar}{2m}\nabla\ln\rho.
\]
\end{lemma}

Mean acceleration:

\[
a = \frac12(D_+D_- + D_-D_+)X.
\]

\begin{theorem}
If $ma=-\nabla V$ holds in mean-square sense,
then $\psi=\sqrt{\rho}e^{iS/\hbar}$ satisfies
\[
i\hbar\partial_t\psi
=
-\frac{\hbar^2}{2m}\Delta\psi + V\psi.
\]
\end{theorem}

Proof uses martingale representation and Doob–Meyer decomposition.

\section{Spin-1/2: Stochastic Variational Derivation of Pauli}

Let $\Psi:\mathbb{R}^3\to\mathbb{C}^2$.

Define density and spin:

\[
\rho=\Psi^\dagger\Psi,
\quad
s^i=\Psi^\dagger\sigma^i\Psi.
\]

Pauli current:

\[
j=
\frac{\hbar}{m}\Im(\Psi^\dagger\nabla\Psi)
-\frac{e}{m}A\rho
+\frac{\hbar}{2m}\nabla\times s.
\]

Define drift $v=j/\rho$.

Stochastic action functional:

\[
\mathcal{A}
=
\mathbb{E}\int dt
\left[
\frac{m}{2}(v^2+u^2)
- e\phi
+ \frac{e}{2m}s\cdot B
\right].
\]

\begin{theorem}
Stationarity of $\mathcal{A}$ yields the Pauli equation:
\[
i\hbar\partial_t\Psi
=
\left[
\frac{1}{2m}(\sigma\cdot(-i\hbar\nabla-eA))^2
+ e\phi
\right]\Psi.
\]
\end{theorem}

Key identity:
\[
(\sigma\cdot\Pi)^2
=
\Pi^2 - e\hbar\sigma\cdot B.
\]

Spin arises from rotational correction to drift
and magnetic coupling in stochastic Lagrangian.

\[

\[

\quad
d\Xi_t = T_a\circ dW^a_t.

\[
\mathcal{L}
=
\frac{\hbar}{2m}\Delta_x

sector yields Pauli spinors.
\end{theorem}

Thus spin is internal Brownian motion on compact Lie group.
